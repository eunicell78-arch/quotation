📋 Supabase 테이블 설정 (최초 1회 실행)

Supabase 프로젝트 → SQL Editor에서 아래 SQL을 실행하세요. anon key가 코드에 포함되어 있으므로 누구나 URL을 알면 접근 가능합니다 (A안). 

-- confirmed_quotes 테이블 생성
CREATE TABLE IF NOT EXISTS public.confirmed_quotes (
  id         uuid        PRIMARY KEY DEFAULT gen_random_uuid(),
  user_key   text        NOT NULL,
  quote_id   text,
  payload    jsonb       NOT NULL,
  created_at timestamptz DEFAULT now()
);

-- 인덱스
CREATE INDEX IF NOT EXISTS idx_cq_user_key  ON public.confirmed_quotes (user_key);
CREATE INDEX IF NOT EXISTS idx_cq_quote_id  ON public.confirmed_quotes (quote_id);

-- RLS 비활성화 (보안 강제 없음, A안)
ALTER TABLE public.confirmed_quotes DISABLE ROW LEVEL SECURITY;

-- anon 역할에 CRUD 권한 부여
GRANT SELECT, INSERT, UPDATE, DELETE
  ON public.confirmed_quotes TO anon;
